Qts Pratiques:

1) et 2)
class Voiture{
    private string $marque;
    private string $modele;
    private int $vitesse;

    public function __construct($marque, $modele, $vitesse){
        $this->marque = $marques;
        $this->modele = $modele;
        $this->vitesse = $vitesse;
    }

    public function accelerer(){
        return $this->vitesse++;
    }
}
